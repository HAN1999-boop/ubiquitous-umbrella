JavaScript 基础知识练习

1. 变量定义与数据类型

// 变量定义（var/let/const）
let userName = "张三"; // 字符串类型
const age = 20;        // 数字类型（常量，不可修改）
var isStudent = true;  // 布尔类型
let hobby = ["编程", "阅读", "运动"]; // 数组类型
let info = {            // 对象类型
    name: userName,
    age: age,
    isStudent: isStudent
};

// 打印变量类型
console.log("userName 类型:", typeof userName);
console.log("age 类型:", typeof age);
console.log("isStudent 类型:", typeof isStudent);
console.log("hobby 类型:", typeof hobby); // 数组typeof返回"object"
console.log("info 类型:", typeof info);

2. 表达式运算

// 算术表达式
let a = 10, b = 3;
let sum = a + b;
let sub = a - b;
let mul = a * b;
let div = a / b;
let mod = a % b; // 取余

// 比较表达式
let isGreater = a > b;
let isEqual = a === b;

// 逻辑表达式
let logicResult = (a > 5) && (b < 5);

console.log("算术运算结果：", sum, sub, mul, div, mod);
console.log("比较运算结果：", isGreater, isEqual);
console.log("逻辑运算结果：", logicResult);

3. 流程控制语句

// 条件判断：if-else
let score = 85;
if (score >= 90) {
    console.log("成绩优秀");
} else if (score >= 80) {
    console.log("成绩良好");
} else if (score >= 60) {
    console.log("成绩及格");
} else {
    console.log("成绩不及格");
}

// 循环语句：for循环
console.log("for循环打印1-5：");
for (let i = 1; i <= 5; i++) {
    console.log(i);
}

// 循环语句：while循环
console.log("while循环打印1-3：");
let j = 1;
while (j <= 3) {
    console.log(j);
    j++;
}

// 多分支：switch
let day = 3;
switch (day) {
    case 1: console.log("星期一"); break;
    case 2: console.log("星期二"); break;
    case 3: console.log("星期三"); break;
    default: console.log("其他日期");
}

4. 函数定义与调用

// 普通函数
function add(num1, num2) {
    return num1 + num2;
}

// 箭头函数（ES6+）
const multiply = (num1, num2) => num1 * num2;

// 带默认参数的函数
function greet(name = "访客") {
    return `你好，${name}！`;
}

// 调用函数
console.log("add(5, 8) =", add(5, 8));
console.log("multiply(4, 6) =", multiply(4, 6));
console.log(greet("小明"));
console.log(greet()); // 使用默认参数
